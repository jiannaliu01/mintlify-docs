Healthcare organizations process thousands of patient forms daily. This tutorial shows you how to automatically extract structured clinical data from PDFs using Cardinal's schema extraction API.

**What you'll build:** A system that takes a patient intake form PDF and extracts structured data ready for EHR integration.

---

## Step 1: Design Your Clinical Data Schema

First, define what data you want to extract. Here's a JSON schema for common clinical information:

```json
{
  "$schema": "https://json-schema.org/draft/2020-12/schema",
  "title": "ClinicalData",
  "type": "object",
  "properties": {
    "patient_info": {
      "type": "object",
      "properties": {
        "mrn": {
          "type": "string",
          "description": "Medical record number"
        },
        "patient_name": {
          "type": "string",
          "description": "Full patient name"
        },
        "dob": {
          "type": "string",
          "format": "date",
          "description": "Date of birth in YYYY-MM-DD format"
        },
        "insurance_id": {
          "type": "string",
          "description": "Insurance identification number"
        }
      },
      "required": ["patient_name", "dob"]
    },
    "clinical_findings": {
      "type": "object",
      "properties": {
        "chief_complaint": {
          "type": "string",
          "description": "Primary reason for visit"
        },
        "medications": {
          "type": "array",
          "items": {
            "type": "object",
            "properties": {
              "name": {
                "type": "string",
                "description": "Medication name"
              },
              "dosage": {
                "type": "string",
                "description": "Medication dosage"
              }
            },
            "required": ["name"]
          }
        },
        "vital_signs": {
          "type": "object",
          "properties": {
            "blood_pressure": {
              "type": "string",
              "description": "Blood pressure reading"
            },
            "heart_rate": {
              "type": "number",
              "description": "Heart rate in BPM"
            }
          }
        }
      }
    }
  },
  "required": ["patient_info", "clinical_findings"]
}
```

## Step 2: Call Cardinal's API

Send your PDF and schema to Cardinal's `/process` endpoint:

```javascript
async function extractClinicalData(pdfUrl) {
  const schema = {
    "$schema": "https://json-schema.org/draft/2020-12/schema",
    "title": "ClinicalData",
    "type": "object",
    "properties": {
      "patient_info": {
        "type": "object",
        "properties": {
          "mrn": {"type": "string", "description": "Medical record number"},
          "patient_name": {"type": "string", "description": "Full patient name"},
          "dob": {"type": "string", "format": "date", "description": "Date of birth"},
          "insurance_id": {"type": "string", "description": "Insurance ID"}
        },
        "required": ["patient_name", "dob"]
      },
      "clinical_findings": {
        "type": "object",
        "properties": {
          "chief_complaint": {"type": "string", "description": "Primary reason for visit"},
          "medications": {
            "type": "array",
            "items": {
              "type": "object",
              "properties": {
                "name": {"type": "string", "description": "Medication name"},
                "dosage": {"type": "string", "description": "Medication dosage"}
              },
              "required": ["name"]
            }
          },
          "vital_signs": {
            "type": "object",
            "properties": {
              "blood_pressure": {"type": "string", "description": "Blood pressure"},
              "heart_rate": {"type": "number", "description": "Heart rate in BPM"}
            }
          }
        }
      }
    },
    "required": ["patient_info", "clinical_findings"]
  };

  const response = await fetch('https://api.cardinal.com/process', {
    method: 'POST',
    headers: { 
      'Content-Type': 'application/json',
      'Authorization': 'Bearer YOUR_API_KEY'
    },
    body: JSON.stringify({
      fileUrl: pdfUrl,
      schema: JSON.stringify(schema)
    })
  });

  return await response.json();
}
```

## Step 3: Parse the Response

Cardinal returns the extracted data as a JSON string in `schema_result`. Parse it to get your structured data:

```javascript
// Call the extraction function
const result = await extractClinicalData('https://your-clinic.com/patient-form.pdf');

// Parse the extracted data
const clinicalData = JSON.parse(result.schema_result);

console.log('Extracted clinical data:', clinicalData);
```

## Step 4: Use the Structured Data

Now you can use this structured data in your healthcare workflow:

```javascript
// Validate required fields
function validateClinicalData(data) {
  if (!data.patient_info?.patient_name) {
    throw new Error('Patient name is required');
  }
  if (!data.patient_info?.dob) {
    throw new Error('Date of birth is required');
  }
  return true;
}

// Send to EHR system
async function sendToEHR(clinicalData) {
  validateClinicalData(clinicalData);
  
  const ehrPayload = {
    patient: {
      name: clinicalData.patient_info.patient_name,
      mrn: clinicalData.patient_info.mrn,
      dateOfBirth: clinicalData.patient_info.dob,
      insuranceId: clinicalData.patient_info.insurance_id
    },
    encounter: {
      chiefComplaint: clinicalData.clinical_findings.chief_complaint,
      medications: clinicalData.clinical_findings.medications,
      vitals: clinicalData.clinical_findings.vital_signs
    }
  };

  const response = await fetch('https://your-ehr-system.com/api/patients', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(ehrPayload)
  });

  return response.json();
}

// Put it all together
async function processPatientForm(pdfUrl) {
  try {
    // Step 1: Extract data from PDF
    const extractionResult = await extractClinicalData(pdfUrl);
    const clinicalData = JSON.parse(extractionResult.schema_result);
    
    // Step 2: Send to EHR
    const ehrResult = await sendToEHR(clinicalData);
    
    console.log('Successfully processed patient form:', ehrResult);
    return ehrResult;
  } catch (error) {
    console.error('Error processing patient form:', error);
    throw error;
  }
}
```
